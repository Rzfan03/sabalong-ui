### 1. Install the Package

Install **sabalong-ui** and its peer dependencies via npm:

```bash
npm install sabalong-ui
```
add this code in global.css

```css
@import "tailwindcss";

@source "../../node_modules/sabalong-ui/dist";
```


### Example Usage Components

```tsx
import { AlertGreen, Button, Badge } from "sabalong-ui";

export default function ExamplePage() {
  return (
    <div className="p-6 space-y-4">
      <AlertGreen message="SabalongUI is ready to use!" />
      
      <div className="flex items-center gap-2">
        <Button label="Action Button" onClick={() => console.log('Clicked!')} />
        <Badge/>
      </div>
    </div>
  );
}
```
